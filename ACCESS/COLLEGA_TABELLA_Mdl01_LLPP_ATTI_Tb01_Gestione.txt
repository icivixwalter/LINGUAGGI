    On Error Resume Next

    ' e poi riscotruito il collegamento con
    ' TrasferisciDatabase
    ' e cioè RICOSTUIRISCE TRUISCE IL COLELGAMENTO trasferisce database. Se invece esiste la tabella  ATTI DI IMPEGNO La stessa
    ' prima elimina la tabella collegata gia esistente

    ' c:\GESTIONI\GESTIONE_LLPP\25_GESTIONE_LLPP\LLPP_ARCHVI_MDB\LLPP_GESTIONE\LLPP_ATTI_Tb01_GESTIONE.mdb


COLLEGA_Mcr_LLPP_ATTI_Tb01_Gestione_Exit:

COLLEGA_Mcr_LLPP_ATTI_Tb01_Gestione_Err:

    Resume COLLEGA_Mcr_LLPP_ATTI_Tb01_Gestione_Exit

End Function
